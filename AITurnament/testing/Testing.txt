
        public void Testing()
        {

            Random rnd = new Random();
            int SampleSize = 10;

            string totalTime = "";
            totalTime += MyStopwatch.Measure(() =>
            {

                while (SampleSize > 0)
                {
                    // 100 millioner * 10 samples, spil i alt *fÃ¸r
                    int rounds = 100;
                    string timeSpend = "";
                    int player1Wins = 0;
                    int player2Wins = 0;
                    int ties = 0;

                    timeSpend += MyStopwatch.Measure(() =>
                    {
                        while (rounds > 0)
                        {
                            TicTacToe ttt = new TicTacToe();

                            while (ttt.gameRunning)
                            {
                                ttt.MakeMove(rnd.Next(0, 3), rnd.Next(0, 3));
                            }

                            if (ttt.winner == 1)
                                player1Wins++;
                            else if (ttt.winner == 2)
                                player2Wins++;
                            else
                                ties++;

                            rounds--;
                        }
                    });
                    string dataLog = "Time spend: " + timeSpend + ". p1 wins: " + player1Wins + ". p2 wins: " + player2Wins + ". Ties: " + ties;
                    MyConsole.WriteLine(dataLog);
                    SampleSize--;
                }
            }, 1);
            MyConsole.WriteLine("Total run time: " + totalTime);
        }
